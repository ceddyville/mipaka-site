Quick Start

Drop-in code for any language

curl
Python
JavaScript 
# List all countries
curl https://api.mipaka.dev/api/v1/countries/

# Kenya counties
curl "https://api.mipaka.dev/api/v1/divisions/?country=KE&level=1"

# Search by historical name
curl "https://api.mipaka.dev/api/v1/divisions/?name=Léopoldville"

# Export as CSV
curl "https://api.mipaka.dev/api/v1/divisions/export/?country=KE" -o kenya.csv
import requests

# List Kenya counties
response = requests.get(
    "https://api.mipaka.dev/api/v1/divisions/",
    params={"country": "KE", "level": 1}
)
counties = response.json()["results"]

for county in counties:
    print(f"{county['name']} — {county['children_count']} constituencies")

# Search historical names
names = requests.get(
